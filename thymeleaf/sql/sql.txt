CREATE TABLE `sys_role` (
  `id` bigint NOT NULL AUTO_INCREMENT,
  `created` datetime(3) DEFAULT CURRENT_TIMESTAMP(3),
  `deleted` datetime(3) DEFAULT NULL,
  `updated` datetime(3) DEFAULT CURRENT_TIMESTAMP(3) ON UPDATE CURRENT_TIMESTAMP(3),
  `role_name` varchar(63) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL,
  `description` varchar(63) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL,
  `status` varchar(20) DEFAULT NULL,
  PRIMARY KEY (`id`) USING BTREE
)

CREATE TABLE `sys_permission` (
  `id` bigint NOT NULL AUTO_INCREMENT,
  `created` datetime(3) DEFAULT CURRENT_TIMESTAMP(3),
  `deleted` datetime(3) DEFAULT NULL,
  `updated` datetime(3) DEFAULT CURRENT_TIMESTAMP(3) ON UPDATE CURRENT_TIMESTAMP(3),
  `permission_url` varchar(63) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL,
  `description` varchar(63) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL,
  `status` varchar(20) DEFAULT NULL,
  PRIMARY KEY (`id`) USING BTREE
)


CREATE TABLE `role_permission` (
  `id` bigint NOT NULL AUTO_INCREMENT,
  `created` datetime(3) DEFAULT CURRENT_TIMESTAMP(3),
  `deleted` datetime(3) DEFAULT NULL,
  `updated` datetime(3) DEFAULT CURRENT_TIMESTAMP(3) ON UPDATE CURRENT_TIMESTAMP(3),
  `role_id` varchar(63) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL,
  `permission_id` varchar(63) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL,
  `status` varchar(20) DEFAULT NULL,
  PRIMARY KEY (`id`) USING BTREE
)

CREATE TABLE `user_role` (
  `id` bigint NOT NULL AUTO_INCREMENT,
  `created` datetime(3) DEFAULT CURRENT_TIMESTAMP(3),
  `deleted` datetime(3) DEFAULT NULL,
  `updated` datetime(3) DEFAULT CURRENT_TIMESTAMP(3) ON UPDATE CURRENT_TIMESTAMP(3),
  `role_id` varchar(63) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL,
  `user_id` varchar(63) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL,
  `status` varchar(20) DEFAULT NULL,
  PRIMARY KEY (`id`) USING BTREE
)
